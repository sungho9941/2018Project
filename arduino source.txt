#include <Wire.h>
#include <LIDARLite.h>
#include <SoftwareSerial.h>

SoftwareSerial BTSerial(2, 3); //블루투스의 Tx, Rx핀을 2번 3번핀으로 설정

#define RED 5
#define YELLOW 7
#define BLUE 9

LIDARLite lidarLite;
int cal_cnt = 0;
int red[3] = {76, 160, 255};
int yellow[3] = {130, 195, 255};
//int yellow[3] = {130, 178, 255};
int blue[3] = {76, 178, 255};
int brightState[3] = {2, 2, 2};

void setup() {
  Serial.begin(9600);  // sets the serial port to 9600
  BTSerial.begin(9600);

  BTSerial.begin(9600);
  lidarLite.begin(0, true); // Set configuration to default and I2C to 400 kHz
  lidarLite.configure(0); // Change this number to try out alternate configurations

  pinMode(BLUE, OUTPUT);
  pinMode(RED, OUTPUT);
  pinMode(YELLOW, OUTPUT);
}
void loop() {
  int dist;
  delay(50);

  // At the beginning of every 100 readings,
  // take a measurement with receiver bias correction
  if ( cal_cnt == 0 ) {
    dist = lidarLite.distance();      // With bias correction
  } else {
    dist = lidarLite.distance(false); // Without bias correction
  }

  // Increment reading counter
  cal_cnt++;
  cal_cnt = cal_cnt % 100;

  // Display distance
  Serial.print(dist);
  Serial.println(" cm");

  if ( dist > 2 && dist <= 800) {
    //빨간색 켜짐  0~30
    analogWrite(YELLOW, 0);
    analogWrite(BLUE, 0);      // 동시에 파란색 LED 꺼짐
    analogWrite(RED, red[brightState[0]]);      // 동시에 빨간색 LED는 켜짐
    BTSerial.write('3');
  }
  else if (dist > 800 && dist <= 1600) {
    // 노란색 켜짐 30~60
    analogWrite(RED, 0);      // 동시에 빨간색 LED 꺼짐
    analogWrite(BLUE, 0);      // 동시에 파란색 LED 꺼짐
    analogWrite(YELLOW, yellow[brightState[1]]);      // 동시에 빨간색 LED는 켜짐
    BTSerial.write('2');
  }
  else if (dist < 2 || dist > 1600 && dist <= 2400) {
    // 파란색 켜짐 30~60
    analogWrite(RED, 0);      // 동시에 빨간색 LED 꺼짐
    analogWrite(YELLOW, 0);     // 동시에 초록색 LED 켜짐
    analogWrite(BLUE, blue[brightState[2]]);      // 동시에 빨간색 LED는 켜짐
    BTSerial.write('1');
  }
  else {
    //파란색을 다 꺼지게 하고!!!
    analogWrite(RED, 0);      // 동시에 빨간색 LED는 꺼짐
    analogWrite(BLUE, 0);     // 동시에 파란색 LED는 꺼짐
    analogWrite(YELLOW, 0);      // 동시에 초록색 LED는 꺼짐
  }

  if (Serial.available()) {    //시리얼모니터에 입력된 데이터가 있다면
    //char view = Serial.read();
//    Serial.println(view);
    BTSerial.write(Serial.read());
  }

  if (BTSerial.available()) { //블루투스에서 넘어온 데이터가 있다면
    int view;
    view = BTSerial.read();
    Serial.println(view); //시리얼모니터에 데이터를 출력
    
    if (view == 12) {
      brightState[2] = 2;
    }
    else if (view == 11) {
      brightState[2] = 1;
    }
    else if (view == 10) {
      brightState[2] = 0;
    }
    else if (view == 22) {
      brightState[1] = 2;
    }
    else if (view == 21) {
      brightState[1] = 1;
    }
    else if (view == 20) {
      brightState[1] = 0;
    }
    else if (view == 32) {
      brightState[0] = 2;
    }
    else if (view == 31) {
      brightState[0] = 1;
    }
    else if (view == 30) {
      brightState[0] = 0;
    }
  }
}
